        <Tab.Navigator>
            <Tab.Screen name='Home' component={HomeTab} />

        </Tab.Navigator>
      </NavigationContainer>
   <NavigationContainer>



   <!-- Tab Navigation Code -->
      <Tab.Navigator screenOptions={{
